
####animationscript function main#####
# data/chars/mary/mary.txt
########################################
void main()
{
    int frame = getlocalvar("frame");
    int animhandle = getlocalvar("animhandle");
    if(animhandle==7)
    {
        if(frame==0)
        {
            spawnbind("zoomin", 0, 0, 0);
        }
        if(frame==9)
        {
            spawnbind("zoomout", 0, 0, 0);
        }
        return;
    }

}
