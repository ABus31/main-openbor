anim idle
@script
	void self = getlocalvar("self"); //Get calling entity.
	int  x = getentityproperty(self, "x");
	int  z = getentityproperty(self, "z");
	int  Vx = getlocalvar("Vx"+self);
	int  Vz = getlocalvar("Vz"+self);
	int  Dir = getlocalvar("D"+self);

	changeentityproperty(self, "velocity", Vx, Vz);
	changeentityproperty(self, "direction", Dir);

        if (z <= openborconstant("PLAYER_MIN_Z")){
          setlocalvar("Vz"+self, 1); 
        } else if (z >= openborconstant("PLAYER_MAX_Z")){
          setlocalvar("Vz"+self, -1); 
        }

        if ( (x >= openborvariant("xpos") + 380) && (Dir==1) ){
          setlocalvar("D"+self,0);
          setlocalvar("Vx"+self,-2);
        } else if ( (x <= openborvariant("xpos") - 60) && (Dir==0) ){
          setlocalvar("D"+self,1);
          setlocalvar("Vx"+self,2);
        }
@end_script
